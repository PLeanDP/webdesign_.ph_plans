CREATE DATABASE IF NOT EXISTS webdesign;
USE webdesign;

CREATE TABLE IF NOT EXISTS admin
	(
	id INT(10) NOT NULL AUTO_INCREMENT,
	username CHAR(20) NOT NULL,
	password CHAR(20) NOT NULL,
	PRIMARY KEY (id)
	);


CREATE TABLE IF NOT EXISTS designer
	(
	id INT(10) NOT NULL AUTO_INCREMENT,
	fname CHAR(20) NOT NULL,
	mname CHAR(20) NOT NULL,
	lname CHAR(20) NOT NULL,
	contact_num CHAR(15) NULL,
	e_add CHAR(30) NOT NULL,
	location TEXT,
	comments TEXT,
	status CHAR(5) NOT NULL,
	PRIMARY KEY (id)
	);


CREATE TABLE IF NOT EXISTS designer_account
	(
	id INT(10),
	username CHAR(20) NOT NULL,
	password CHAR(20) NOT NULL
	);

CREATE TABLE IF NOT EXISTS designer_sites
	(
	id INT(10) NOT NULL AUTO_INCREMENT,
	designer_id INT(10) NOT NULL,
	file_name CHAR(30),
	site_name CHAR(20),
	site_desc text,
	PRIMARY KEY (id)
	);


CREATE TABLE IF NOT EXISTS client
	(
	id INT(10) NOT NULL AUTO_INCREMENT,
	name char(150) NOT NULL,
	location TEXT,
	contact_no CHAR(15),
	email_add CHAR(30),
	PRIMARY KEY (id)
	);

CREATE TABLE IF NOT EXISTS request
	(
	client_id char(10),
	designer_id char(50)
	);





alter table REQUEST add constraint foreign key(client_id) references CLIENT(username);
alter table REQUEST add constraint foreign key(des_id) references DESIGNER(username);
alter table DES_WEB_IMG add constraint foreign key(des_id) references DESIGNER(username);
