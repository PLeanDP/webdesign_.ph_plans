create database WEBDESIGN;
use WEBDESIGN;

create table DESIGNER_ACCOUNT
(
	username varchar(20) not null,
	password varchar(20) not null
);

create table DESIGNER
(
	username varchar(20) not null,
	des_fname varchar(30) not null,
	des_lname varchar(30) not null,
	des_mname varchar(30) null,
	contact_no varchar(13) null,
	email_add varchar(30) null,
	location varchar(50) null,
	comments varchar(200) null
);

create table CLIENT
(
	username varchar(20) not null,
	client_fname varchar(30) not null,
	client_lname varchar(30) not null,
	client_mname varchar(30) null,
	contact_no varchar(13) null,
	email_add varchar(30) null
);

create table REQUEST
(
	client_id varchar(20) not null,
	des_id varchar(20) not null,
	comments varchar(200) null
);

create table DES_WEB_IMG
(
	des_id varchar(20) not null,
	image_id int not null,
	image_des varchar(100) null,
	image_loc varchar(50) not null
);

alter table DESIGNER_ACCOUNT add constraint primary key (username);
alter table DESIGNER add constraint primary key(username);
alter table CLIENT add constraint primary key(username);
alter table REQUEST add constraint primary key(client_id,des_id);
alter table DES_WEB_IMG add constraint primary key(des_id,image_id);
alter table REQUEST add constraint foreign key(client_id) references CLIENT(username);
alter table REQUEST add constraint foreign key(des_id) references DESIGNER(username);
alter table DES_WEB_IMG add constraint foreign key(des_id) references DESIGNER(username);
